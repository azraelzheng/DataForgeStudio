## 三、API 接口设计 (API Design)

### 3.1 项目管理 API (Project Management)

| 操作 | 方法 | 路径 | 请求体 | 响应 |
|------|------|------|--------|------|
| 创建项目 | POST | `/report/go-view/project/create` | `{name, picUrl, content, status, remark}` | `{code, msg, data: projectId}` |
| 更新项目 | PUT | `/report/go-view/project/update` | `{id, name, picUrl, content, status, remark}` | `{code, msg, data: true}` |
| 删除项目 | DELETE | `/report/go-view/project/delete?id=xxx` | - | `{code, msg, data: true}` |
| 获取项目 | GET | `/report/go-view/project/get?id=xxx` | - | `{code, msg, data: project}` |
| 我的项目列表 | GET | `/report/go-view/project/my-page` | Query: pageNo, pageSize | `{code, msg, data: {list, total}}` |

### 3.2 数据查询 API (Data Query)

| 操作 | 方法 | 路径 | 请求体 | 响应 |
|------|------|------|--------|------|
| SQL 查询 | POST | `/report/go-view/data/get-by-sql` | `{sql}` | `{dimensions, source}` |
| HTTP 查询 | POST | `/report/go-view/data/get-by-http` | `{url, method, params}` | 动态数据 |
| **报表查询** | POST | `/report/go-view/data/get-by-report` | `{reportId, params}` | `{dimensions, source}` |
| **报表列表** | GET | `/report/go-view/data/report-list` | - | 报表元数据列表 |

**数据响应格式**：

```json
{
  "code": 0,
  "msg": "success",
  "data": {
    "dimensions": ["name", "value", "category"],
    "source": [
      {"name": "产品A", "value": 100, "category": "类别1"},
      {"name": "产品B", "value": 200, "category": "类别2"}
    ]
  }
}
```

### 3.3 数据源方式对比 (Data Source Comparison)

| 方式 | 优点 | 缺点 | 适用场景 |
|------|------|------|----------|
| **SQL 查询** | 简单、灵活 | 安全风险、需懂数据库结构 | 临时查询、简单报表 |
| **HTTP 查询** | 可调用外部系统 | 需要开发接口 | 跨系统集成 |
| **报表调用** | 复用业务逻辑、统一管理、安全 | 需要注册报表 | 企业级应用、多图表复用 |

**推荐**：在 Vue3 实现中优先使用 **报表调用** 方式，这样可以：
- 复用已有报表逻辑
- 避免直接写 SQL
- 统一权限控制
- 便于数据缓存

#### 5.2.2 拖拽 Hook (useDragDrop.ts)

```typescript
import { ref } from 'vue'

export function useDragDrop(addComponent: (type: string, position: Position) => void) {
  const isDragging = ref(false)
  const dragType = ref<string | null>(null)

  const handleDragStart = (type: string) => {
    isDragging.value = true
    dragType.value = type
  }

  const handleDragEnd = () => {
    isDragging.value = false
    dragType.value = null
  }

  const handleDrop = (event: DragEvent, canvasRect: DOMRect) => {
    if (!dragType.value) return

    const position = {
      x: event.clientX - canvasRect.left,
      y: event.clientY - canvasRect.top
    }

    addComponent(dragType.value, position)
    handleDragEnd()
  }

  return {
    isDragging,
    dragType,
    handleDragStart,
    handleDragEnd,
    handleDrop
  }
}
```

### 5.3 状态管理 (State Management - Pinia Store)

```typescript
// src/store/dashboard.ts
import { defineStore } from 'pinia'
import type { DashboardConfig, ComponentConfig } from '@/types/dashboard'

export const useDashboardStore = defineStore('dashboard', {
  state: () => ({
    // 当前编辑的大屏配置
    currentDashboard: null as DashboardConfig | null,

    // 选中的组件 ID
    selectedComponentId: null as string | null,

    // 历史记录（用于撤销/重做）
    history: [] as DashboardConfig[],
    historyIndex: -1,

    // 是否已修改
    isDirty: false
  }),

  getters: {
    components: (state) => state.currentDashboard?.components ?? [],
    selectedComponent: (state) => {
      if (!state.currentDashboard || !state.selectedComponentId) return null
      return state.currentDashboard.components.find(c => c.id === state.selectedComponentId)
    }
  },

  actions: {
    // 加载大屏配置
    async loadDashboard(id: string) {
      const response = await fetch(`/report/go-view/project/get?id=${id}`)
      const result = await response.json()
      this.currentDashboard = JSON.parse(result.data.content)
    },

    // 保存大屏配置
    async saveDashboard() {
      if (!this.currentDashboard) return

      const response = await fetch('/report/go-view/project/update', {
        method: 'PUT',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          id: this.currentDashboard.id,
          name: this.currentDashboard.name,
          content: JSON.stringify(this.currentDashboard)
        })
      })

      this.isDirty = false
      return response.json()
    },

    // 添加组件
    addComponent(type: string, position: { x: number; y: number }) {
      if (!this.currentDashboard) return

      const newComponent: ComponentConfig = {
        id: `comp_${Date.now()}`,
        type,
        title: this.getComponentTitle(type),
        layout: {
          x: position.x,
          y: position.y,
          width: 300,
          height: 200,
          zIndex: this.getMaxZIndex() + 1,
          rotation: 0
        },
        dataSource: {
          type: 'static',
          staticData: this.getDefaultData(type)
        },
        styles: {},
        animation: {
          enter: 'fadeIn',
          duration: 500,
          delay: 0
        },
        interactions: []
      }

      this.currentDashboard.components.push(newComponent)
      this.selectedComponentId = newComponent.id
      this.saveHistory()
    },

    // 更新组件
    updateComponent(id: string, updates: Partial<ComponentConfig>) {
      const component = this.currentDashboard?.components.find(c => c.id === id)
      if (component) {
        Object.assign(component, updates)
        this.saveHistory()
        this.isDirty = true
      }
    },

    // 删除组件
    deleteComponent(id: string) {
      if (!this.currentDashboard) return

      const index = this.currentDashboard.components.findIndex(c => c.id === id)
      if (index > -1) {
        this.currentDashboard.components.splice(index, 1)
        this.selectedComponentId = null
        this.saveHistory()
        this.isDirty = true
      }
    },

    // 撤销
    undo() {
      if (this.historyIndex > 0) {
        this.historyIndex--
        this.currentDashboard = JSON.parse(JSON.stringify(this.history[this.historyIndex]))
      }
    },

    // 重做
    redo() {
      if (this.historyIndex < this.history.length - 1) {
        this.historyIndex++
        this.currentDashboard = JSON.parse(JSON.stringify(this.history[this.historyIndex]))
      }
    },

    // 保存历史记录
    saveHistory() {
      if (!this.currentDashboard) return

      // 截断后面的历史
      this.history = this.history.slice(0, this.historyIndex + 1)

      // 添加当前状态
      this.history.push(JSON.parse(JSON.stringify(this.currentDashboard)))
      this.historyIndex = this.history.length - 1
    },

    // 辅助方法
    getComponentTitle(type: string): string {
      const titles: Record<string, string> = {
        'chart-line': '折线图',
        'chart-bar': '柱状图',
        'chart-pie': '饼图',
        'text': '文本',
        'image': '图片'
      }
      return titles[type] || type
    },

    getMaxZIndex(): number {
      if (!this.currentDashboard) return 0
      return Math.max(0, ...this.currentDashboard.components.map(c => c.layout.zIndex))
    },

    getDefaultData(type: string): any {
      // 返回组件的默认数据
      if (type.startsWith('chart-')) {
        return [
          { name: '一月', value: 100 },
          { name: '二月', value: 200 },
          { name: '三月', value: 150 }
        ]
      }
      return null
    }
```

### 6.2 数据源处理 (Data Source Handler)

```typescript
// src/composables/useDataSource.ts
import { ref, watch } from 'vue'

export function useDataSource(config: Ref<ComponentConfig>) {
  const data = ref<any[]>([])
  const loading = ref(false)
  const error = ref<string | null>(null)

  const fetchData = async () => {
    const { dataSource } = config.value

    switch (dataSource.type) {
      case 'static':
        data.value = dataSource.staticData || []
        break

      case 'api':
        loading.value = true
        try {
          const response = await fetch(dataSource.apiUrl!, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(dataSource.requestParams)
          })
          const result = await response.json()
          data.value = result.data?.source || result.data || []
        } catch (e) {
          error.value = '数据加载失败'
        } finally {
          loading.value = false
        }
        break

      case 'sql':
        loading.value = true
        try {
          const response = await fetch('/report/go-view/data/get-by-sql', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ sql: dataSource.sql })
          })
          const result = await response.json()
          data.value = result.data?.source || []
        } catch (e) {
          error.value = 'SQL 查询失败'
        } finally {
          loading.value = false
        }
        break
    }
  }

  // 初始加载
  fetchData()

  // 定时刷新
  if (config.value.dataSource.refreshInterval) {
    const interval = setInterval(fetchData, config.value.dataSource.refreshInterval * 1000)
    onUnmounted(() => clearInterval(interval))
  }

  return { data, loading, error, refresh: fetchData }
}
```

```java
@Service
public class ReportDataServiceImpl implements ReportDataService {

    @Resource
    private ReportDataSourceMapper reportDataSourceMapper;

    @Resource
    private ApplicationContext applicationContext;

    // 本地缓存
    private final Cache<String, GoViewDataRespVO> dataCache =
        Caffeine.newBuilder()
            .expireAfterWrite(5, TimeUnit.MINUTES)
            .build();

    @Override
    public GoViewDataRespVO getDataByReport(String reportCode, Map<String, Object> params) {
        // 1. 查询报表配置
        ReportDataSourceDO dataSource = reportDataSourceMapper.selectByCode(reportCode);
        if (dataSource == null || dataSource.getStatus() != 1) {
            throw new ServiceException(REPORT_NOT_EXISTS);
        }

        // 2. 检查缓存
        String cacheKey = buildCacheKey(reportCode, params);
        if (dataSource.getCacheSeconds() > 0) {
            GoViewDataRespVO cached = dataCache.getIfPresent(cacheKey);
            if (cached != null) {
                return cached;
            }
        }

        // 3. 反射调用 Service 方法
        Object service = applicationContext.getBean(dataSource.getServiceBean());
        Object result = invokeMethod(service, dataSource.getMethodName(), params);

        // 4. 转换为 GoView 格式
        GoViewDataRespVO respVO = convertToGoViewData(result, dataSource.getResultMapping());

        // 5. 缓存结果
        if (dataSource.getCacheSeconds() > 0) {
            dataCache.put(cacheKey, respVO);
        }

        return respVO;
    }

    @Override
    public List<ReportDataSourceRespVO> getAvailableReports() {
        return reportDataSourceMapper.selectList(
            new LambdaQueryWrapper<ReportDataSourceDO>()
                .eq(ReportDataSourceDO::getStatus, 1)
                .orderByAsc(ReportDataSourceDO::getCategory)
                .orderByAsc(ReportDataSourceDO::getName)
        ).stream().map(this::toRespVO).collect(Collectors.toList());
    }

    @Override
    public List<ReportParamDefinition> getParamDefinitions(String reportCode) {
        ReportDataSourceDO dataSource = reportDataSourceMapper.selectByCode(reportCode);
        if (dataSource == null || StringUtils.isEmpty(dataSource.getParamsSchema())) {
            return Collections.emptyList();
        }
        // 解析 JSON Schema 返回参数定义
        return parseParamsSchema(dataSource.getParamsSchema());
    }

    private Object invokeMethod(Object service, String methodName, Map<String, Object> params) {
        try {
            Method method = service.getClass().getMethod(methodName, Map.class);
            return method.invoke(service, params);
        } catch (Exception e) {
            throw new ServiceException(REPORT_INVOKE_FAILED, e.getMessage());
        }
    }

    private GoViewDataRespVO convertToGoViewData(Object result, String resultMapping) {
        GoViewDataRespVO respVO = new GoViewDataRespVO();

        // 根据结果类型转换
        if (result instanceof List) {
            List<?> list = (List<?>) result;
            if (!list.isEmpty()) {
                // 从第一条数据提取维度
                Object first = list.get(0);
                if (first instanceof Map) {
                    respVO.setDimensions(new ArrayList<>(((Map<?, ?>) first).keySet()));
                } else {
                    // 使用 Java Bean 属性
                    respVO.setDimensions(getBeanPropertyNames(first.getClass()));
                }
            }
            respVO.setSource((List<Map<String, Object>>) list);
        }
        // 可扩展其他类型转换

        return respVO;
    }
}
```

#### 7.3.3 更新 useDataSource Hook

```typescript
// src/composables/useDataSource.ts
import { ref, watch, onUnmounted, type Ref } from 'vue'
import type { ComponentConfig } from '@/types/dashboard'
import { getDataByReport, getDataBySql, getDataByApi } from '@/api/dashboard'

export function useDataSource(config: Ref<ComponentConfig>) {
  const data = ref<any[]>([])
  const loading = ref(false)
  const error = ref<string | null>(null)

  const fetchData = async () => {
    const { dataSource } = config.value
    if (!dataSource) return

    loading.value = true
    error.value = null

    try {
      let result

      switch (dataSource.type) {
        case 'static':
          data.value = dataSource.staticData || []
          break

        case 'report':
          // 调用报表数据源
          result = await getDataByReport({
            reportId: dataSource.reportId,
            params: dataSource.reportParams
          })
          data.value = result.data?.source || []
          break

        case 'sql':
          result = await getDataBySql({ sql: dataSource.sql })
          data.value = result.data?.source || []
          break

        case 'api':
          result = await getDataByApi({
            url: dataSource.apiUrl,
            method: dataSource.apiMethod || 'POST',
            params: dataSource.requestParams
          })
          data.value = result.data?.source || result.data || []
          break

        default:
          data.value = []
      }

      // 应用字段映射
      if (dataSource.dataMapping && data.value.length > 0) {
        data.value = applyDataMapping(data.value, dataSource.dataMapping)
      }

    } catch (e: any) {
      error.value = e.message || '数据加载失败'
      console.error('数据加载失败:', e)
    } finally {
      loading.value = false
    }
  }

  // 应用字段映射
  const applyDataMapping = (source: any[], mapping: DataMapping) => {
    return source.map(item => ({
      _x: item[mapping.xField],
      _y: item[mapping.yField],
      _series: mapping.seriesField ? item[mapping.seriesField] : undefined,
      ...item
    }))
  }

  // 初始加载
  fetchData()

  // 定时刷新
  let interval: ReturnType<typeof setInterval> | null = null
  if (config.value.dataSource?.refreshInterval && config.value.dataSource.refreshInterval > 0) {
    interval = setInterval(fetchData, config.value.dataSource.refreshInterval * 1000)
  }

  // 监听数据源变化
  watch(
    () => config.value.dataSource,
    () => fetchData(),
    { deep: true }
  )

  // 清理
  onUnmounted(() => {
    if (interval) clearInterval(interval)
  })

  return { data, loading, error, refresh: fetchData }
}
```

### 9.2 全屏展示模式 (Fullscreen Mode)

```typescript
// src/composables/useFullscreen.ts
export function useFullscreen() {
  const isFullscreen = ref(false)

  const enterFullscreen = () => {
    const el = document.documentElement
    if (el.requestFullscreen) {
      el.requestFullscreen()
      isFullscreen.value = true
    }
  }

  const exitFullscreen = () => {
    if (document.exitFullscreen) {
      document.exitFullscreen()
      isFullscreen.value = false
    }
  }

  const toggleFullscreen = () => {
    isFullscreen.value ? exitFullscreen() : enterFullscreen()
  }

  return { isFullscreen, enterFullscreen, exitFullscreen, toggleFullscreen }
}
```
